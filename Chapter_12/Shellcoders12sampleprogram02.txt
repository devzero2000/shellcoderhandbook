#include <alpha/regdef.h>
#include <alpha/pal.h>
     .text
     .arch     generic
     .align 4
     .globl  main
     .ent      main
main:
     bis  zero, zero, a0       #argument to setuid(), uid=0
     addq zero, 0x17, v0       #setuid system call number.
     PAL_callsys               #trap to kernel mode.
